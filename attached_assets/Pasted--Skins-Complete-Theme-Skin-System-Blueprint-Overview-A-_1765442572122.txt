# Skins - Complete Theme/Skin System Blueprint

## Overview
A comprehensive, production-ready skin/theme system for React + TypeScript + Tailwind CSS applications. Supports 300+ themes across 15 categories with persistent user preferences, optional watermarks/logos, and a beautiful accordion-based selector UI.

---

## Architecture

```
client/
  src/
    contexts/
      ThemeContext.tsx       # React context provider
    data/
      themes.ts              # Theme definitions (all themes)
    components/
      ThemeSelector.tsx      # UI component for selecting themes
```

## Step 1: Theme Context Provider

Create `client/src/contexts/ThemeContext.tsx`:

```typescript
import { createContext, useContext, useState, ReactNode } from "react";
import { allThemes } from "@/data/themes";

export interface ThemeColors {
  primary: string;      // Gradient: "from-blue-900 via-slate-800 to-blue-900"
  secondary: string;    // Gradient: "from-blue-600 to-purple-600"
  accent: string;       // Single color: "bg-blue-600"
  background: string;   // Background: "bg-slate-900"
  cardBg: string;       // Card background: "bg-slate-800"
  textPrimary: string;  // Primary text: "text-white"
  textSecondary: string; // Secondary text: "text-slate-300"
}

export interface Theme {
  id: string;           // Unique identifier: "bills", "cherry-blossom"
  name: string;         // Display name: "Buffalo Bills", "Cherry Blossom"
  colors: ThemeColors;
  watermark?: string;   // Optional logo/image URL
  category: ThemeCategory;
}

export type ThemeCategory = 
  | "classic" 
  | "nfl" | "mlb" | "nba" | "nhl" | "mls" 
  | "wsl" | "epl" | "laliga" | "bundesliga" | "seriea" | "ligue1" 
  | "college" | "golf" | "nature";

interface ThemeContextType {
  currentTheme: Theme;
  setTheme: (themeId: string) => void;
  availableThemes: Theme[];
}

const ThemeContext = createContext<ThemeContextType | undefined>(undefined);

// Storage key - change this for your app
const STORAGE_KEY = "app_theme";

export function ThemeProvider({ children }: { children: ReactNode }) {
  const [currentTheme, setCurrentTheme] = useState<Theme>(() => {
    const savedThemeId = localStorage.getItem(STORAGE_KEY);
    return allThemes.find(t => t.id === savedThemeId) || allThemes[0];
  });

  const setTheme = (themeId: string) => {
    const theme = allThemes.find(t => t.id === themeId);
    if (theme) {
      setCurrentTheme(theme);
      localStorage.setItem(STORAGE_KEY, themeId);
    }
  };

  return (
    <ThemeContext.Provider value={{ currentTheme, setTheme, availableThemes: allThemes }}>
      {children}
    </ThemeContext.Provider>
  );
}

export function useTheme() {
  const context = useContext(ThemeContext);
  if (!context) {
    throw new Error("useTheme must be used within ThemeProvider");
  }
  return context;
}
```

## Step 3: Theme Selector Component

Create `client/src/components/ThemeSelector.tsx`:

```typescript
import { useState } from "react";
import { useTheme, ThemeCategory } from "@/contexts/ThemeContext";
import { Badge } from "@/components/ui/badge";
import { 
  Accordion, 
  AccordionContent, 
  AccordionItem, 
  AccordionTrigger 
} from "@/components/ui/accordion";
import { CheckCircle2, Palette, Sparkles, Star, Crown } from "lucide-react";

const categoryLabels: Record<string, { label: string; icon: string; color: string }> = {
  classic: { label: "Classic Themes", icon: "Palette", color: "from-slate-600 to-slate-800" },
  nfl: { label: "NFL Teams", icon: "Football", color: "from-green-600 to-green-800" },
  mlb: { label: "MLB Teams", icon: "Baseball", color: "from-red-600 to-red-800" },
  nba: { label: "NBA Teams", icon: "Basketball", color: "from-orange-600 to-orange-800" },
  nhl: { label: "NHL Teams", icon: "Hockey", color: "from-blue-600 to-blue-800" },
  mls: { label: "MLS Teams", icon: "Soccer", color: "from-emerald-600 to-emerald-800" },
  nature: { label: "Nature & Scenery", icon: "Trees", color: "from-teal-600 to-teal-800" }
};

const categoryOrder = ["classic", "nfl", "mlb", "nba", "nhl", "mls", "nature"];

export function ThemeSelector() {
  const { currentTheme, setTheme, availableThemes } = useTheme();
  const [expandedCategories, setExpandedCategories] = useState<string[]>([]);

  // Group themes by category
  const themesByCategory = availableThemes.reduce((acc, theme) => {
    if (!acc[theme.category]) {
      acc[theme.category] = [];
    }
    acc[theme.category].push(theme);
    return acc;
  }, {} as Record<string, typeof availableThemes>);

  const sortedCategories = categoryOrder.filter(cat => themesByCategory[cat]?.length > 0);

  return (
    <div className="space-y-4 bg-gradient-to-br from-slate-900 via-slate-800 to-slate-900 rounded-2xl p-4 border border-slate-700/50">
      {/* Header */}
      <div className="flex items-center gap-3">
        <div className="bg-gradient-to-br from-blue-500 to-purple-600 p-2.5 rounded-full">
          <Palette className="h-5 w-5 text-white" />
        </div>
        <div>
          <h2 className="text-lg font-bold text-white flex items-center gap-2">
            Theme Gallery
            <Sparkles className="h-4 w-4 text-yellow-400" />
          </h2>
          <p className="text-xs text-slate-400">
            {availableThemes.length}+ premium themes
          </p>
        </div>
      </div>

      {/* Current Theme Display */}
      <div className="rounded-xl bg-gradient-to-r from-blue-900/50 to-purple-900/50 border border-blue-500/30 p-3">
        <div className="flex items-center gap-3">
          <div 
            className={`w-12 h-12 rounded-lg bg-gradient-to-br ${currentTheme.colors.primary} border-2 border-white/20`}
            style={{
              backgroundImage: currentTheme.watermark ? `url(${currentTheme.watermark})` : undefined,
              backgroundSize: "cover",
              backgroundPosition: "center",
              backgroundBlendMode: "overlay"
            }}
          />
          <div className="flex-1">
            <div className="flex items-center gap-2">
              <Crown className="h-4 w-4 text-yellow-400" />
              <span className="text-sm font-bold text-white">{currentTheme.name}</span>
            </div>
            <p className="text-xs text-blue-300">Currently Active</p>
          </div>
          <CheckCircle2 className="h-5 w-5 text-green-400" />
        </div>
      </div>

      {/* Accordion Categories */}
      <Accordion 
        type="multiple" 
        value={expandedCategories}
        onValueChange={setExpandedCategories}
        className="space-y-2"
      >
        {sortedCategories.map((category) => {
          const themes = themesByCategory[category];
          const catInfo = categoryLabels[category];
          const hasCurrentTheme = themes.some(t => t.id === currentTheme.id);
          
          return (
            <AccordionItem key={category} value={category} className="border-0">
              <AccordionTrigger 
                className={`px-3 py-2.5 rounded-xl bg-gradient-to-r ${catInfo.color} border border-white/10
                  ${hasCurrentTheme ? 'ring-2 ring-yellow-400/50' : ''}`}
              >
                <div className="flex items-center gap-3 w-full">
                  <span className="text-sm font-bold text-white">{catInfo.label}</span>
                  <Badge className="ml-auto mr-2 bg-white/20 text-white border-white/30 text-xs">
                    {themes.length}
                  </Badge>
                  {hasCurrentTheme && <Star className="h-4 w-4 text-yellow-400 fill-yellow-400" />}
                </div>
              </AccordionTrigger>
              
              <AccordionContent className="pt-3 pb-1">
                <div className="overflow-x-auto pb-2">
                  <div className="flex gap-2" style={{ minWidth: "max-content" }}>
                    {themes.map((theme) => {
                      const isSelected = currentTheme.id === theme.id;
                      return (
                        <button
                          key={theme.id}
                          onClick={() => setTheme(theme.id)}
                          className={`relative flex-shrink-0 w-20 rounded-xl overflow-hidden transition-all
                            ${isSelected 
                              ? 'ring-2 ring-yellow-400 scale-105' 
                              : 'hover:scale-105 border border-slate-600/50'
                            }`}
                          data-testid={`theme-${theme.id}`}
                        >
                          <div 
                            className={`h-14 bg-gradient-to-br ${theme.colors.primary}`}
                            style={{
                              backgroundImage: theme.watermark ? `url(${theme.watermark})` : undefined,
                              backgroundSize: "cover",
                              backgroundPosition: "center",
                              backgroundBlendMode: "overlay"
                            }}
                          >
                            {isSelected && (
                              <div className="absolute top-1 right-1">
                                <div className="bg-yellow-400 rounded-full p-0.5">
                                  <CheckCircle2 className="h-3 w-3 text-slate-900" />
                                </div>
                              </div>
                            )}
                          </div>
                          <div className="p-1.5 bg-slate-800/90">
                            <p className="text-[10px] font-medium text-white truncate text-center">
                              {theme.name.substring(0, 12)}
                            </p>
                          </div>
                        </button>
                      );
                    })}
                  </div>
                </div>
              </AccordionContent>
            </AccordionItem>
          );
        })}
      </Accordion>
    </div>
  );
}
```

    <ThemeProvider>

## Step 5: Apply Theme Colors in Components

Use the `useTheme` hook to apply theme colors anywhere:

```typescript
import { useTheme } from "@/contexts/ThemeContext";

function MyComponent() {
  const { currentTheme } = useTheme();
  
  return (
    <div className={`bg-gradient-to-br ${currentTheme.colors.primary}`}>
      <div className={currentTheme.colors.background}>
        <div className={currentTheme.colors.cardBg}>
          <h1 className={currentTheme.colors.textPrimary}>Title</h1>
          <p className={currentTheme.colors.textSecondary}>Subtitle</p>
          <button className={currentTheme.colors.accent}>Action</button>
        </div>
      </div>
    </div>
  );
}
```

## Step 6: Apply Theme to Root Layout

For full-app theming, apply to your root layout:

```typescript
function RootLayout({ children }) {
  const { currentTheme } = useTheme();
  
  return (
    <div className={`min-h-screen ${currentTheme.colors.background} ${currentTheme.colors.textPrimary}`}>
      {children}
    </div>
  );
}
```

## Storage

Themes persist to localStorage with the key defined in ThemeContext (default: `app_theme`). Change this key to avoid conflicts between apps.

## Key Features

1. **300+ themes** - Sports teams, nature scenes, classic colors
2. **15 categories** - NFL, NBA, MLB, NHL, MLS, EPL, La Liga, Bundesliga, Serie A, Ligue 1, WSL, College, Golf, Nature, Classic
3. **Watermark support** - Team logos and background images
4. **Persistent preferences** - Saves to localStorage
5. **Accordion UI** - Organized, collapsible category groups
6. **Horizontal scroll** - Easy theme browsing within categories
7. **Active theme indicator** - Visual feedback for current selection
8. **Type-safe** - Full TypeScript support

---

## Notes for Implementation

1. Replace the storage key (`STORAGE_KEY`) with your app's unique prefix
2. Add/remove categories based on your needs
3. The watermark URLs point to external images - ensure they're stable or host your own
4. All colors use Tailwind CSS classes - adjust for your Tailwind configuration
5. The helper function `createTeamTheme` automatically infers background/text colors from the primary gradient
